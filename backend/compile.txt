# delete local branch (safe - only if merged)
git branch -d feature/search_flights_by_airline

# if branch not merged and you want to force delete (use with care)
git branch -D feature/search_flights_by_airline

# delete remote branch
git push origin --delete feature/search_flights_by_airline

git revert <commithash> --no-edit

git fetch --all
git branch -r           # list all remote branches

# quick heuristic: show remote branches and show which are not tracked locally
git fetch origin
git remote show origin

# per-line attribution (who changed each line)
git blame --line-porcelain path/to/register.jsp
# show full patch/details of last commit that changed the file
git log -1 -p -- path/to/register.jsp

# show the last commit with full diff
git show HEAD
